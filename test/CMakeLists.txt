cmake_minimum_required (VERSION 2.8)
project(IOHexperimenter-test)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -std=c++11 -Wall  -lIOH")
find_package(GTest REQUIRED)
find_package(Threads REQUIRED)
include_directories(${GTEST_INCLUDE_DIRS})
add_executable(PBOTETS test-pbo-problem.cpp)
add_executable(BBOBTEST test-pbo-problem.cpp)
target_link_libraries(PBOTETS ${GTEST_BOTH_LIBRARIES})
target_link_libraries(PBOTETS ${CMAKE_THREAD_LIBS_INIT})
target_link_libraries(BBOBTEST ${GTEST_BOTH_LIBRARIES})
target_link_libraries(BBOBTESTm ${CMAKE_THREAD_LIBS_INIT})
add_test(Test PBOTEST)
add_test(Test BBOBTEST)
enable_testing()
